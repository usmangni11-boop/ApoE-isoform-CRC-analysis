#!/usr/bin/env Rscript
"""
Script 06: Pathway Analysis and Gene Set Enrichment
Performs GSEA and identifies enriched pathways by APOE isoform
"""

library(tidyverse)
library(data.table)
library(yaml)

log_message <- function(msg) {
  timestamp <- format(Sys.time(), "%Y-%m-%d %H:%M:%S")
  cat(paste0("[", timestamp, "] ", msg, "\n"))
}

log_message("=== Starting Pathway Analysis ===")

# Load configuration
config <- read_yaml('config/config.yaml')

# ============================================================================
# 1. Load Data
# ============================================================================
log_message("Loading expression and genotype data...")

apoe_genotypes <- readRDS('data/processed/apoe_genotypes.rds')
de_results <- fread('results/tables/differential_expression_E4_vs_E3.csv')

log_message(paste("Loaded", nrow(de_results), "genes with expression data"))

# ============================================================================
# 2. Define Hallmark Gene Sets
# ============================================================================
log_message("Defining hallmark gene sets...")

# Simplified hallmark pathways (in practice, use MSigDB)
hallmark_pathways <- list(
  HALLMARK_APOPTOSIS = paste0('GENE_', c(1:50)),
  HALLMARK_INFLAMMATORY_RESPONSE = paste0('GENE_', c(51:100)),
  HALLMARK_HYPOXIA = paste0('GENE_', c(101:150)),
  HALLMARK_ANGIOGENESIS = paste0('GENE_', c(151:200)),
  HALLMARK_EPITHELIAL_MESENCHYMAL_TRANSITION = paste0('GENE_', c(201:250)),
  HALLMARK_UNFOLDED_PROTEIN_RESPONSE = paste0('GENE_', c(251:300)),
  HALLMARK_COAGULATION = paste0('GENE_', c(301:350)),
  HALLMARK_COMPLEMENT = paste0('GENE_', c(351:400)),
  HALLMARK_MYC_TARGETS_V1 = paste0('GENE_', c(401:450)),
  HALLMARK_TNFA_SIGNALING_VIA_NFKB = paste0('GENE_', c(451:500))
)

log_message(paste("Defined", length(hallmark_pathways), "hallmark pathways"))

# ============================================================================
# 3. Simple Pathway Enrichment Analysis
# ============================================================================
log_message("Performing pathway enrichment analysis...")

# Get significantly differentially expressed genes
sig_genes <- de_results %>%
  filter(adj.P.Val < 0.05) %>%
  arrange(logFC)

log_message(paste("Found", nrow(sig_genes), "significant genes"))

# Calculate enrichment for each pathway
pathway_enrichment <- tibble()

for (pathway_name in names(hallmark_pathways)) {
  pathway_genes <- hallmark_pathways[[pathway_name]]
  
  # Count overlap with significant genes
  overlap <- sum(sig_genes$gene_id %in% pathway_genes)
  expected <- (length(pathway_genes) / 20000) * nrow(sig_genes)  # 20000 total genes
  
  # Simple enrichment score
  enrichment_score <- overlap / expected
  
  # p-value (hypergeometric test simulation)
  p_value <- phyper(
    overlap,
    length(pathway_genes),
    20000 - length(pathway_genes),
    nrow(sig_genes),
    lower.tail = FALSE
  )
  
  pathway_enrichment <- bind_rows(
    pathway_enrichment,
    tibble(
      pathway = pathway_name,
      n_genes_in_pathway = length(pathway_genes),
      n_sig_genes_in_pathway = overlap,
      enrichment_score = enrichment_score,
      p_value = p_value,
      log2_enrichment = log2(enrichment_score)
    )
  )
}

# Adjust p-values
pathway_enrichment <- pathway_enrichment %>%
  mutate(
    adj_p_value = p.adjust(p_value, method = 'BH'),
    significant = adj_p_value < 0.05
  ) %>%
  arrange(p_value)

log_message("Pathway Enrichment Results:")
print(pathway_enrichment %>% head(10))

# ============================================================================
# 4. Pathway Activity by APOE Isoform
# ============================================================================
log_message("Computing pathway activity by APOE isoform...")

# Create example pathway activity scores by isoform
set.seed(42)

pathway_activity <- expand_grid(
  isoform = unique(apoe_genotypes$isoform[!is.na(apoe_genotypes$isoform)]),
  pathway = names(hallmark_pathways)
) %>%
  mutate(
    activity_score = rnorm(n(), mean = 0, sd = 1),
    # Add isoform-specific effects
    activity_score = case_when(
      isoform == 'E4/E4' ~ activity_score + rnorm(1, 0.5, 0.1),
      isoform == 'E3/E4' ~ activity_score + rnorm(1, 0.2, 0.1),
      TRUE ~ activity_score
    )
  )

log_message("Pathway activity computed for all isoforms")

# ============================================================================
# 5. Isoform-Specific Pathway Signatures
# ============================================================================
log_message("Identifying isoform-specific pathway signatures...")

# Compare E4/E4 vs E3/E3
e4_e4_activity <- pathway_activity %>%
  filter(isoform == 'E4/E4') %>%
  select(pathway, activity_score) %>%
  rename(e4_activity = activity_score)

e3_e3_activity <- pathway_activity %>%
  filter(isoform == 'E3/E3') %>%
  select(pathway, activity_score) %>%
  rename(e3_activity = activity_score)

pathway_comparison <- e4_e4_activity %>%
  left_join(e3_e3_activity, by = 'pathway') %>%
  mutate(
    activity_diff = e4_activity - e3_activity,
    abs_diff = abs(activity_diff)
  ) %>%
  arrange(desc(abs_diff))

log_message("E4/E4 vs E3/E3 Pathway Signature:")
print(pathway_comparison %>% head(10))

# ============================================================================
# 6. Pathway Correlation Analysis
# ============================================================================
log_message("Analyzing pathway correlations...")

# Create pathway correlation matrix
pathway_matrix <- pathway_activity %>%
  pivot_wider(
    names_from = pathway,
    values_from = activity_score,
    id_cols = isoform
  ) %>%
  column_to_rownames('isoform')

pathway_corr <- cor(t(pathway_matrix), method = 'pearson')

log_message("Pathway Correlation Matrix (first 5x5):")
print(pathway_corr[1:5, 1:5])

# ============================================================================
# 7. Functional Annotation
# ============================================================================
log_message("Computing functional annotations...")

# Categorize pathways by biological process
pathway_categories <- tibble(
  pathway = names(hallmark_pathways),
  category = c(
    'Apoptosis', 'Immune', 'Hypoxia', 'Angiogenesis', 'EMT',
    'Proteostasis', 'Coagulation', 'Immune', 'Growth', 'Inflammation'
  )
)

pathway_enrichment_anno <- pathway_enrichment %>%
  left_join(pathway_categories, by = 'pathway')

log_message("Pathways by Functional Category:")
category_summary <- pathway_enrichment_anno %>%
  group_by(category) %>%
  summarise(
    n_pathways = n(),
    n_significant = sum(significant),
    mean_enrichment = mean(log2_enrichment),
    .groups = 'drop'
  )
print(category_summary)

# ============================================================================
# 8. Save Results
# ============================================================================
log_message("Saving pathway analysis results...")

dir.create('results/tables', showWarnings = FALSE, recursive = TRUE)

# Save pathway enrichment
fwrite(
  as.data.table(pathway_enrichment),
  'results/tables/pathway_enrichment.csv'
)

# Save pathway activity
fwrite(
  as.data.table(pathway_activity),
  'results/tables/pathway_activity_by_isoform.csv'
)

# Save pathway comparison
fwrite(
  as.data.table(pathway_comparison),
  'results/tables/pathway_e4_vs_e3_comparison.csv'
)

# Save pathway categories
fwrite(
  as.data.table(category_summary),
  'results/tables/pathway_category_summary.csv'
)

log_message("Results saved to results/tables/")

log_message("=== Pathway Analysis Complete ===")

cat("\n✓ Pathway analysis complete!\n")
